/**
 * @input A : Integer
 * 
 * @Output string array. You need to malloc memory. 
 *   Fill in len1 as number of strings. Make sure every string ends with null character
 */
char** fizzBuzz(int A, int *len1) {
    int i;
for(i=1;i<=A;i++)
if(i%3==0 && i%5==0)
printf("FizzBuzz ");
else if(i%3==0)
printf("Fizz ");
else if(i%5==0)
printf("Buzz ");
else
printf("%d ",i);
return;
}
